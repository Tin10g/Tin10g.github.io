class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        std::vector<int> res={0,0};
        for(int i=0; i<nums.size(); i++){
            int num1 = nums[i];
            for(int j=i+1; j<nums.size();j++){
                if(num1 + nums[j] == target){
                    res[0] = i;
                    res[1] = j;
                    break;
                }
            }
        }
        return res;
    }
};

class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        string res = "";
        string tmp = "";
        string s = strs[0];
        for(int i=0; i<s.size(); i++){
            tmp = s[i];
            for(int j=0; j<strs.size(); j++){
                string s1 = strs[j];
                if(s1[i] == ' ' || s1[i] != s[i]) return res;
            }
            res = res + tmp;
        }
        return res;
    }
};

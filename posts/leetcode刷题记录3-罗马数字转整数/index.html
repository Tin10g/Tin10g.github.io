class Solution {
public:
    int romanToInt(std::string s) {
        std::unordered_map<char, int> dict = {{'I', 1}, {'V', 5}, {'X', 10}, {'L', 50}, {'C', 100}, {'D', 500}, {'M', 1000}};
        int num = 0;
        int before = 0;
        int now = 0;
        
        for(int i = s.size() - 1; i >= 0; i--) {
            now = dict[s[i]];
            if(now < before) {
                num -= now;
            } else {
                num += now;
            }
            before = now;
        }
        return num;     
    }
};

class Solution {
public:
    int romanToInt(string s) {
        if (s.size() == 0) 
            return 0;
        switch (s[0]) {
            case 'I':
                if (s.size() > 1) {
                    switch (s[1]) {
                        case 'V':
                            return 4 + romanToInt(s.substr(2));
                        case 'X':
                            return 9 + romanToInt(s.substr(2));
                        default:
                            return 1 + romanToInt(s.substr(1));
                    }
                } else {
                    return 1;
                }
            case 'V':
                return 5 + romanToInt(s.substr(1));
            case 'X':
                if (s.size() > 1) {
                    switch (s[1]) {
                        case 'L':
                            return 40 + romanToInt(s.substr(2));
                        case 'C':
                            return 90 + romanToInt(s.substr(2));
                        default:
                            return 10 + romanToInt(s.substr(1));
                    }
                } else {
                    return 10;
                }
            case 'L':
                return 50 + romanToInt(s.substr(1));
            case 'C':
                if (s.size() > 1) {
                    switch (s[1]) {
                        case 'D':
                            return 400 + romanToInt(s.substr(2));
                        case 'M':
                            return 900 + romanToInt(s.substr(2));
                        default:
                            return 100 + romanToInt(s.substr(1));
                    }
                } else {
                    return 100;
                }
            case 'D':
                return 500 + romanToInt(s.substr(1));
            case 'M':
                return 1000 + romanToInt(s.substr(1));
            default:
                return -1;
        }
    }
};

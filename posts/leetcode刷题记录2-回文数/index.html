class Solution {
public:
    bool isPalindrome(int x) {
        string s1 = std::to_string(x);
        string s2 = s1;
        reverse(s2.begin(), s2.end());
        if(s1 == s2) return true;
        else return false;
    }
};

class Solution {
public:
    bool isPalindrome(int x) {
        if (x<0 || (x!=0 && x%10 ==0)) return false;
        int revertNum = 0;
        while(revertNum < x){
            revertNum = revertNum*10 + x%10;
            x = x/10;
        }
        if(revertNum == x || revertNum/10 == x ) return true;
        return false;
    }
};
